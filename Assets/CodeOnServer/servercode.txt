
# A very simple Flask Hello World app for you to get started with...

from flask import Flask, request
import json

app = Flask(__name__)

@app.route('/')
def hello_world():
    return 'Hello from mcast!'


@app.route("/api/saveboxes",methods=['POST'])
def passdata():
    print("saving data on server")

    content = request.get_json()

    fh = open("/home/gld62bmcast/mysite/datafile.txt","w")
    fh.write(json.dumps(content))
    fh.close()

    return "Data Saved Correctly"


@app.route("/api/getboxes",methods=['GET'])
def getdata():
    fh = open("/home/gld62bmcast/mysite/datafile.txt","r")
    savedText = fh.read()
    fh.close()

    return savedText
